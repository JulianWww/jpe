>>> a = [[1,2],[3,4]]
>>> b = copy.copy(a)
# std copy lib copy

>>> a[0] = 10
# in this case b is [[1,2],[3,4]]
# so it works

however

>>> a[1][0] = 10
# in this case b is [[1,2],[10,4]] but in a true copy it schould be [[1,2], [3,4]]
# if we use this patch
>>> a = [[1,2],[3,4]]
>>> c = jpe.utils.copy.copy(a)
